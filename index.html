let arr = [[1, 2, '^^', 5], [3, '^^', '^^'], ['^^', 4, '^^'], ['^^', 5, 6]];

function countCats(array) {
    let count = 0;
    for (let x = 0; x < array.length; x++) {
        for (let y = 0; y < array[x].length; y++) {
            if (array[x][y] === '^^')
                count++
        }
    }
    return count;
}
